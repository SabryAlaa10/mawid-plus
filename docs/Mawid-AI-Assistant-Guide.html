خياران للنشر

الخيار A — Python FastAPI على Railway/Render (موصى به)

  • تحكم كامل في الكود والأداء
  • تشغيل نماذج embedding محلياً (أسرع)
  • إمكانية استخدام Ollama للتشغيل المحلي بدون تكلفة
  • سهولة الصيانة والتطوير

الخيار B — Supabase Edge Function تنادي Python API

  • الـ Edge Function كـ proxy يمرر الطلبات لـ Python
  • أكثر تعقيداً بدون فائدة حقيقية
  • غير موصى به — استخدم Python مباشرة
التوصية: استخدم الخيار A (FastAPI مباشرة). التطبيق يتصل بالـ API عبر HTTPS، والـ API يتصل بـ Supabase وبنموذج اللغة. أبسط وأقوى.

ملف الاتصال: app/database.py

python 
import os
from supabase import create_client, Client
from dotenv import load_dotenv

load_dotenv()

supabase: Client = create_client(
    os.getenv("SUPABASE_URL"),
    os.getenv("SUPABASE_SERVICE_ROLE_KEY")  # service role للعمليات الخلفية
)


# ──────── استعلام الأطباء حسب التخصص ────────
def get_doctors_by_specialty(specialty: str, limit: int = 3):
    """جلب أفضل الأطباء حسب التخصص مرتبين بالتقييم"""
    result = supabase.table("doctors") \
        .select("id, full_name, specialty, rating, review_count, "
                "clinic_address, avatar_url, available_days, "
                "start_time, end_time, slot_duration_minutes, "
                "consultation_fee_sar") \
        .ilike("specialty", f"%{specialty}%") \
        .order("rating", desc=True) \
        .limit(limit) \
        .execute()

    if not result.data:
        # fallback: جلب أفضل الأطباء عموماً
        result = supabase.table("doctors") \
            .select("*") \
            .order("rating", desc=True) \
            .limit(limit) \
            .execute()

    return result.data


# ──────── جلب الأوقات المتاحة ────────
def get_available_slots(doctor_id: str, date: str):
    """حساب الأوقات المتاحة لطبيب في يوم معين"""
    # 1. جلب مواعيد عمل الطبيب
    doc = supabase.table("doctors") \
        .select("start_time, end_time, slot_duration_minutes") \
        .eq("id", doctor_id) \
        .single() \
        .execute()

    if not doc.data:
        return []

    # 2. جلب المواعيد المحجوزة
    booked = supabase.table("appointments") \
        .select("time_slot") \
        .eq("doctor_id", doctor_id) \
        .eq("appointment_date", date) \
        .neq("status", "cancelled") \
        .execute()

    booked_slots = {b["time_slot"] for b in (booked.data or [])}

    # 3. حساب كل الأوقات الممكنة
    start_h, start_m = map(int, doc.data["start_time"].split(":")[:2])
    end_h, end_m = map(int, doc.data["end_time"].split(":")[:2])
    duration = doc.data.get("slot_duration_minutes", 30)

    slots = []
    h, m = start_h, start_m
    while h < end_h or (h == end_h and m < end_m):
        time_str = f"{h:02d}:{m:02d}"
        slots.append({
            "time": time_str,
            "available": time_str not in booked_slots
        })
        m += duration
        if m >= 60:
            h += m // 60
            m = m % 60

    return slots


# ──────── حجز موعد ────────
def book_appointment(patient_id: str, doctor_id: str,
                     date: str, time_slot: str):
    """حجز موعد جديد"""
    # التأكد إن الوقت مش محجوز
    existing = supabase.table("appointments") \
        .select("id") \
        .eq("doctor_id", doctor_id) \
        .eq("appointment_date", date) \
        .eq("time_slot", time_slot) \
        .neq("status", "cancelled") \
        .limit(1) \
        .execute()

    if existing.data:
        return {"error": "هذا الموعد محجوز بالفعل"}

    # حساب رقم الطابور
    day_appts = supabase.table("appointments") \
        .select("id", count="exact") \
        .eq("doctor_id", doctor_id) \
        .eq("appointment_date", date) \
        .neq("status", "cancelled") \
        .execute()

    queue_number = (day_appts.count or 0) + 1

    # إنشاء الموعد
    result = supabase.table("appointments").insert({
        "patient_id": patient_id,
        "doctor_id": doctor_id,
        "appointment_date": date,
        "time_slot": time_slot,
        "queue_number": queue_number,
        "status": "scheduled"
    }).execute()

    return {
        "success": True,
        "appointment": result.data[0] if result.data else None,
        "queue_number": queue_number
    }

5 بناء خط أنابيب RAG

ما هو RAG؟

Retrieval-Augmented Generation = بدل ما الـ LLM يخترع إجابات من دماغه، بيدوّر الأول في قاعدة المعرفة الطبية على معلومات مشابهة لسؤال المريض، وبعدين يستخدمها عشان يرد رد دقيق.

سؤال المريض ──► تحويل لـ Embedding ──► بحث في pgvector ──► أقرب 3 نتائج │ ▼ ┌─────────────────────────────┐ │ System Prompt + نتائج RAG │ │ + سؤال المريض │ │ ▼ │ │ LLM يولد الرد │ └─────────────────────────────┘

الخطوة 1: إنشاء جدول kb_chunks في Supabase

شغّل SQL ده في Supabase SQL Editor:

sql 
-- تفعيل pgvector
create extension if not exists vector with schema extensions;

-- جدول أجزاء المعرفة الطبية
create table if not exists public.kb_chunks (
    id uuid primary key default gen_random_uuid(),
    content text not null,
    category text not null default '',
    specialty text,
    severity text default 'متوسط',
    metadata jsonb default '{}',
    embedding vector(384),          -- لـ all-MiniLM-L6-v2
    created_at timestamptz default now()
);

-- فهرس للبحث السريع
create index if not exists idx_kb_embedding
    on public.kb_chunks
    using ivfflat (embedding vector_cosine_ops)
    with (lists = 10);

-- دالة البحث بالتشابه
create or replace function match_kb_chunks(
    query_embedding vector(384),
    match_threshold float default 0.5,
    match_count int default 3
)
returns table (
    id uuid, content text, category text,
    specialty text, metadata jsonb, similarity float
)
language sql stable as $$
    select
        id, content, category, specialty, metadata,
        1 - (embedding <=> query_embedding) as similarity
    from public.kb_chunks
    where 1 - (embedding <=> query_embedding) > match_threshold
    order by embedding <=> query_embedding
    limit match_count;
$$;

الخطوة 3: خدمة RAG — app/services/rag.py

python 
# app/services/rag.py
from sentence_transformers import SentenceTransformer
from app.database import supabase

# تحميل النموذج مرة واحدة عند بدء التطبيق
_model = SentenceTransformer("sentence-transformers/all-MiniLM-L6-v2")


def search_medical_knowledge(query: str, top_k: int = 3):
    """البحث في قاعدة المعرفة الطبية بالتشابه الدلالي"""
    # 1. تحويل سؤال المريض لـ embedding
    query_embedding = _model.encode(query).tolist()

    # 2. البحث في pgvector
    result = supabase.rpc("match_kb_chunks", {
        "query_embedding": query_embedding,
        "match_threshold": 0.4,
        "match_count": top_k
    }).execute()

    return result.data or []


def build_rag_context(query: str) -> str:
    """بناء سياق RAG لإضافته للـ prompt"""
    chunks = search_medical_knowledge(query)

    if not chunks:
        return ""

    context_parts = []
    for chunk in chunks:
        context_parts.append(
            f"[{chunk['category']}] {chunk['content']}"
        )

    return "\n\n".join(context_parts)
لماذا all-MiniLM-L6-v2؟ نموذج خفيف (80MB)، يعمل محلياً بدون API، سريع جداً، ويدعم النصوص العربية بشكل مقبول. لو عايز دقة أعلى للعربي استخدم aubmindlab/bert-base-arabertv2 لكنه أبطأ.

from app.auth import get_current_user

router = APIRouter()


@router.post("/chat", response_model=ChatResponse)
async def chat(req: ChatRequest, user=Depends(get_current_user)):
    if not req.message or not req.message.strip():
        raise HTTPException(400, "الرسالة فارغة")

    # 1. بحث RAG
    rag_context = build_rag_context(req.message)

    # 2. استدعاء LLM
    assistant_message = await call_llm(
        user_message=req.message,
        rag_context=rag_context,
        session_id=req.session_id
    )

    # 3. كشف التخصص تلقائياً
    specialty = detect_specialty(assistant_message)

    # 4. جلب أطباء لو تم كشف تخصص
    doctors = None
    if specialty:
        doctors = get_doctors_by_specialty(specialty)

    # 5. ردود سريعة مقترحة
    quick_replies = generate_quick_replies(
        req.message, assistant_message
    )

    return ChatResponse(
        session_id=req.session_id,
        assistant_message=assistant_message,
        quick_replies=quick_replies,
        detected_specialty=specialty,
        recommended_doctors=doctors,
    )


def generate_quick_replies(user_msg: str, bot_msg: str) -> list[str]:
    """توليد ردود سريعة بناءً على السياق"""
    if "طوارئ" in bot_msg or "فوراً" in bot_msg:
        return ["اتصل بالإسعاف", "اذهب للطوارئ"]
    if "أنصحك" in bot_msg or "مراجعة" in bot_msg:
        return ["اعرض الأطباء", "معلومات أكثر", "سؤال آخر"]
    if "؟" in bot_msg:
        return ["نعم", "لا", "أحياناً"]
    return ["حجز موعد", "معلومات أكثر", "سؤال آخر"]

# app/auth.py
import os
from fastapi import HTTPException, Header
from jose import jwt, JWTError

SUPABASE_JWT_SECRET = os.getenv("SUPABASE_JWT_SECRET")
# تجده في: Supabase Dashboard > Settings > API > JWT Secret


async def get_current_user(
    authorization: str = Header(...)
) -> dict:
    """التحقق من JWT token المرسل من Android"""
    if not authorization.startswith("Bearer "):
        raise HTTPException(401, "غير مصرح")

    token = authorization.replace("Bearer ", "")

    try:
        payload = jwt.decode(
            token,
            SUPABASE_JWT_SECRET,
            algorithms=["HS256"],
            audience="authenticated"
        )
        return {
            "id": payload.get("sub"),
            "email": payload.get("email"),
            "role": payload.get("role"),
        }
    except JWTError:
        raise HTTPException(401, "جلسة منتهية، سجل دخول مجدداً")

python 
# app/services/llm.py
import os
import httpx

SYSTEM_PROMPT = """أنت مساعد طبي ذكي لتطبيق موعد+.
مهمتك: مساعدة المريض في فهم أعراضه وتوجيهه للتخصص الطبي المناسب.

قواعد صارمة:
1. لا تشخّص أمراضاً أبداً — أنت لست طبيباً
2. لا تقترح أدوية أبداً
3. وجّه للتخصص المناسب فقط
4. طوارئ (ألم صدر حاد، صعوبة تنفس، نزيف) → اطلب الذهاب فوراً
5. كن ودوداً ومطمئناً
6. اطرح سؤالاً واحداً فقط للمتابعة
7. عندما تكون مستعداً، قل "أنصحك بمراجعة [التخصص]"
8. رد بالعربي دائماً
9. كن مختصراً (3-5 جمل كحد أقصى)"""


async def call_llm(
    user_message: str,
    rag_context: str = "",
    session_id: str = None
) -> str:
    """استدعاء نموذج اللغة مع سياق RAG"""

    system = SYSTEM_PROMPT
    if rag_context:
        system += f"\n\nمعلومات طبية موثوقة:\n{rag_context}"

    messages = [
        {"role": "system", "content": system},
        {"role": "user", "content": user_message}
    ]

    # ── الخيار 1: OpenRouter ──
    api_key = os.getenv("OPENROUTER_API_KEY")
    if api_key:
        return await _call_openrouter(messages, api_key)

    # ── الخيار 2: Groq ──
    groq_key = os.getenv("GROQ_API_KEY")
    if groq_key:
        return await _call_groq(messages, groq_key)

    raise Exception("لم يتم تعيين مفتاح LLM API")


async def _call_openrouter(messages: list, api_key: str) -> str:
    """استدعاء OpenRouter API"""
    models = [
        "meta-llama/llama-3.2-3b-instruct:free",
        "google/gemma-2-9b-it:free",
        "mistralai/mistral-7b-instruct:free",
    ]

    async with httpx.AsyncClient(timeout=30) as client:
        for model in models:
            try:
                resp = await client.post(
                    "https://openrouter.ai/api/v1/chat/completions",
                    headers={
                        "Authorization": f"Bearer {api_key}",
                        "Content-Type": "application/json",
                    },
                    json={
                        "model": model,
                        "max_tokens": 300,
                        "messages": messages
                    }
                )

                if resp.status_code == 429:
                    continue  # rate limited, جرب الموديل التالي

                data = resp.json()
                content = data.get("choices", [{}])[0] \
                              .get("message", {}) \
                              .get("content")
                if content:
                    return content

            except Exception:
                continue

    raise Exception("كل النماذج فشلت")


async def _call_groq(messages: list, api_key: str) -> str:
    """استدعاء Groq API (أسرع خيار)"""
    async with httpx.AsyncClient(timeout=30) as client:
        resp = await client.post(
            "https://api.groq.com/openai/v1/chat/completions",
            headers={
                "Authorization": f"Bearer {api_key}",
                "Content-Type": "application/json",
            },
            json={
                "model": "llama-3.1-8b-instant",
                "max_tokens": 300,
                "messages": messages
            }
        )

        data = resp.json()
        return data["choices"][0]["message"]["content"]

def detect_specialty(text: str) -> str | None:
    """كشف التخصص الطبي من نص رد المساعد"""
    for specialty, keywords in SPECIALTY_KEYWORDS.items():
        if any(kw in text for kw in keywords):
            return specialty
    return None
